numbers = [1,4,3,2,6,5]
print("Before sorting ",numbers)
numbers.sort()
print("After sorting",numbers)
numbers.reverse()
print("After reversing the list ",numbers)
numbers.append(7)
print("After adding element 7 ",numbers)
numbers.insert(1,11)
print("After inserting 11 in 1st index ",numbers)
numbers.remove(3)
print("After removing element 3",numbers)
numbers.pop(0)
print("After removing element at 0th index ",numbers)
numbers.clear()
print("After clearing all elemenst we get our list to be empty ",numbers)
